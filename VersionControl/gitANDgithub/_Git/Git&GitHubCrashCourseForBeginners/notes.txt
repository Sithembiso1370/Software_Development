git init //initialises a local git repository
git add // add files to Index
git status // Check statusn of the working tree
git commit // commit changes to Index
git push // Push to remote repository
git pull // Pull latest version/changes from remote repository
git clone  // copy a remote repository into my folder


//set project attributes
git config --global user.name "[username]" // sets the user name of the project
git config --global user.name "[username]" // sets the user email of the project



// Staging 

//Adding or updating
git add * or git add . // stage all currently unstaged files in the folder
git add (filename )   // stage file with specified filename
git add .(file extension) // stage file with specified file extension

//Deleting 
git rm --cached (filename,filenames/file ext) // unstages the file but does not delete it from the folder
git rm -f (filename,filenames/file ext) //usntages and deletes the file from the folder



// commiting 

git commit //adds the files to Index
git commit "comment" adds files even unstaged directly to commit/index

// Other commands for the command line interface

touch // creates a new file in a specified directory with a specified name
